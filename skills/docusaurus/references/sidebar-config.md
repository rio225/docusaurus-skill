## Basic Structure

```javascript
// sidebars.js
/** @type {import('@docusaurus/plugin-content-docs').SidebarsConfig} */
const sidebars = {
  docs: [
    'intro',
    'getting-started',
    {
      type: 'category',
      label: 'Guides',
      items: ['guides/setup', 'guides/config'],
    },
  ],
};

export default sidebars;
```

## Item Types

### Doc Item (Shorthand)
```javascript
'intro'  // Links to docs/intro.md
'guides/setup'  // Links to docs/guides/setup.md
```

### Doc Item (Explicit)
```javascript
{
  type: 'doc',
  id: 'intro',
  label: 'Introduction',  // Override sidebar label
  className: 'sidebar-intro',  // Custom CSS class
}
```

### Category
```javascript
{
  type: 'category',
  label: 'Getting Started',
  items: ['installation', 'configuration'],
}
```

### External Link
```javascript
{
  type: 'link',
  label: 'GitHub',
  href: 'https://github.com/org/repo',
  description: 'View source code',  // Tooltip text
  className: 'external-link',
}
```

### HTML (Custom Content)
```javascript
{
  type: 'html',
  value: '<hr class="sidebar-divider" />',
  defaultStyle: true,  // Apply default styling
  className: 'custom-html',
}
```

### Autogenerated
```javascript
{
  type: 'autogenerated',
  dirName: '.',  // Generate from docs root
}

// Or specific directory
{
  type: 'autogenerated',
  dirName: 'api',  // Generate from docs/api/
}
```

## Category Options

### Full Category Configuration
```javascript
{
  type: 'category',
  label: 'Guides',

  // Collapsibility
  collapsed: true,       // Start collapsed (default: true)
  collapsible: true,     // Allow collapse (default: true)

  // Category link (makes label clickable)
  link: {
    type: 'doc',
    id: 'guides/overview',  // Link to existing doc
  },
  // OR
  link: {
    type: 'generated-index',
    title: 'All Guides',
    description: 'Explore our comprehensive guides',
    slug: '/guides',
    keywords: ['guides', 'tutorials'],
    image: '/img/guides-og.png',
  },

  // Styling
  className: 'guides-category',

  // Items
  items: [
    'guides/quick-start',
    'guides/advanced',
    {
      type: 'category',
      label: 'Nested Category',
      items: ['guides/nested/doc1'],
    },
  ],
}
```

### Generated Index Options
```javascript
link: {
  type: 'generated-index',
  title: 'API Reference',           // Page title
  description: 'Complete API docs', // Page description
  slug: '/category/api',            // Custom URL
  keywords: ['api', 'reference'],   // SEO keywords
  image: '/img/api-og.png',         // OG image
}
```

## Autogenerated Sidebars

### Full Autogeneration
```javascript
const sidebars = {
  docs: [
    {
      type: 'autogenerated',
      dirName: '.',
    },
  ],
};
```

### Control Order with Frontmatter
```yaml
# In docs/getting-started.md
---
sidebar_position: 1
---
```

### Control Order with _category_.json
Create `docs/guides/_category_.json`:
```json
{
  "label": "Guides",
  "position": 2,
  "collapsed": false,
  "collapsible": true,
  "className": "guides-category",
  "link": {
    "type": "generated-index",
    "title": "Guides Overview",
    "description": "Step-by-step tutorials"
  },
  "customProps": {
    "description": "Custom property for plugins"
  }
}
```

### Mixed Manual + Auto
```javascript
{
  type: 'category',
  label: 'Tutorials',
  items: [
    'tutorials/intro',  // Manual first item
    {
      type: 'autogenerated',
      dirName: 'tutorials/basics',  // Auto middle
    },
    'tutorials/conclusion',  // Manual last item
  ],
}
```

## Multiple Sidebars

### Define Multiple Sidebars
```javascript
const sidebars = {
  docs: [
    'intro',
    {
      type: 'category',
      label: 'Guides',
      items: ['guides/setup', 'guides/deploy'],
    },
  ],

  api: [
    'api/overview',
    {
      type: 'autogenerated',
      dirName: 'api',
    },
  ],

  community: [
    'community/support',
    'community/contributing',
  ],
};
```

### Link Doc to Specific Sidebar
```yaml
# In docs/api/users.md
---
displayed_sidebar: api
---
```

### Navbar with Multiple Sidebars
```javascript
// docusaurus.config.ts
navbar: {
  items: [
    {
      type: 'docSidebar',
      sidebarId: 'docs',
      label: 'Docs',
    },
    {
      type: 'docSidebar',
      sidebarId: 'api',
      label: 'API',
    },
  ],
}
```

## Common Patterns

### Sidebar with Dividers
```javascript
[
  'intro',
  { type: 'html', value: '<hr/>' },
  {
    type: 'category',
    label: 'Core Concepts',
    items: ['concepts/architecture', 'concepts/plugins'],
  },
  { type: 'html', value: '<hr/>' },
  {
    type: 'link',
    label: 'External Resources',
    href: 'https://example.com',
  },
]
```

### Section Headers (Non-clickable)
```javascript
{
  type: 'html',
  value: '<span class="sidebar-section">GETTING STARTED</span>',
  className: 'sidebar-section-header',
}
```

### Hide Item from Sidebar (Keep in Build)
```yaml
# In frontmatter
---
unlisted: true
---
```

### Conditional Items Based on Environment
```javascript
const sidebars = {
  docs: [
    'intro',
    ...(process.env.NODE_ENV === 'development'
      ? ['internal/dev-notes']
      : []),
    'getting-started',
  ],
};
```

### Versioned Docs Sidebar
```javascript
// sidebars.js handles current version
// versioned_sidebars/version-1.0-sidebars.json for older versions
```
